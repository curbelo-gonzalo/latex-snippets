% \CPlusPlus
% Description: Inserts "C++".
% Arguments: None
% Remarks: None
\newcommand{\CPlusPlus}{%
    C\nolinebreak[4]\hspace{-0.1em}\raisebox{0.35ex}{\footnotesize\textbf{++}}%
}

% \CSharp
% Description: Inserts "C#".
% Arguments: None
% Remarks: None
\newcommand{\CSharp}{%
    C\nolinebreak[4]\hspace{-0.1em}\raisebox{0.35ex}{\footnotesize\textbf{\#{}}}%
}
