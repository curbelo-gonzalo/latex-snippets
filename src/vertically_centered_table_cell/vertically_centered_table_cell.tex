% =============================================================================
% DESCRIPTION:
% This file contains commands to insert vertically centered cells in a table.

% =============================================================================
% PREAMBLE:
% \documentclass{article}

% =============================================================================
% REMARKS:
% This was only tested in documentclass "article". It may or may not work with
%   other classes.
% This works correctly inside the following nested environments:
%
% \begin{table}[ .. ]
    % \begin{tabular}[ ... ] % also works with tabular*
        % \multirow{ ... }{ ... } {
            % \VerticallyCenteredCell{ contents... }
        % }
    % end{tabular}
% end{table}
%
% It is unknown whether this works correctly with other kinds of tables.

% =============================================================================
% CONTENTS:

% \VerticallyCenteredCell
% Desciption: Inserts a vertically centered table cell.
% Arguments:
%   #1 = (Required) Contents of the cell.
% Remarks: None
\newcommand{\VerticallyCenteredCell}[1]{%
    \begin{tabular}{ @{} l @{} }%
        #1%
    \end{tabular}%
}
